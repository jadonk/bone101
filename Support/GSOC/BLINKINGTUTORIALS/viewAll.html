var b = require('bonescript');
    var led2 = "P8_12";
    var led1 = "P8_10";
    var state = 0;

    b.pinMode(led1, 'out');
    b.pinMode(led2, 'out');
    toggleLED = function() {
        state = state ? 0 : 1;
        b.digitalWrite(led1, state);
        b.digitalWrite(led2, state);
    };

    timer = setInterval(toggleLED, 500);

    stopTimer = function() {
        clearInterval(timer);
    };

    setTimeout(stopTimer, 12000);
